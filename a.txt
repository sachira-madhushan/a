const decimal = document.getElementById("decimal");
const octal = document.getElementById("octal");
const hex = document.getElementById("hex");
const form = document.getElementById("form");

function convertDecimal(decimal, base) {
  let rem = 0;
  let divident = decimal;
  let divisor = base;
  let value = "";

  const hexMap = {
    10: "A",
    11: "B",
    12: "C",
    13: "D",
    14: "E",
    15: "F",
  };

  while (divident >= 1) {
    rem = divident % divisor;
    divident = parseInt(divident / divisor);
    if (base === 16 && rem >= 10) {
      rem = hexMap[rem];
    }
    value = rem + value;
  }
  return value;
}

form.addEventListener("submit", (e) => {
  e.preventDefault();
  binary.innerText = convertDecimal(parseInt(decimal.value), 2);
  hex.innerText = convertDecimal(parseInt(decimal.value), 16);
  octal.innerText = convertDecimal(parseInt(decimal.value), 8);
});

form.addEventListener("reset", (e) => {
  binary.innerText = "";
  octal.innerText = "";
  hex.innerText = "";
});

      <label for="marks">Marks</label>
      <input type="number" name="marks" id="marks" min="0" max="100" />

      <br />
      <br />

      <label for="grade">Grade</label>
      <input type="text" name="grade" id="grade" readonly />

      <button type="submit">Calculate</button>
      <button type="reset">Reset</button>
    </form>

    <script>
      const marks = document.getElementById("marks");
      const grade = document.getElementById("grade");
      const form = document.getElementById("form");

      function calculate() {
        const mark = parseInt(marks.value);

        if (mark >= 75) grade.value = "A+";
        else if (mark >= 70 && mark < 75) grade.value = "A";
        else if (mark >= 65 && mark < 70) grade.value = "A-";
        else if (mark >= 60 && mark < 65) grade.value = "B+";
        else if (mark >= 55 && mark < 60) grade.value = "B";
        else if (mark >= 50 && mark < 55) grade.value = "B-";
        else if (mark >= 45 && mark < 50) grade.value = "C+";
        else if (mark >= 40 && mark < 45) grade.value = "C";
        else if (mark < 40) grade.value = "D";
      }

      form.addEventListener("submit", (e) => {
        e.preventDefault();
        calculate();
      });
    </script>
  </body>
</html>
